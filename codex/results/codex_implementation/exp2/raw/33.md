\[
\frac{\frac{1}{a} + \frac{1}{b} + \frac{1}{c}}{3} \geq \sqrt[3]{\frac{1}{abc}}.

\[
\sqrt[3]{\frac{1}{abc}} = \sqrt[3]{1} = 1.
